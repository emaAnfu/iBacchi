		\begin{lstlisting}
		int findMax (float array [], int size);
		int findMax (float array [], int size){
			int index=0;
			float max=array[0];
			for (int i=1; i<size; i++){
				if (array[i]>max){
				max=array[i];
				index=i;
				}
			}
			return index;
		} 
		\end{lstlisting}
